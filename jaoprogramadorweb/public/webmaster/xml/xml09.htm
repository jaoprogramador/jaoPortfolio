<?xml version="1.0"?>
<!DOCTYPE Informe SYSTEM "Ej1.dtd">
<bookmark>
<metainfo>
<titulo>Programacion Java</titulo>
<autor>Juantxu</autor>
<resumen>web de programación en Java</resumen>
</metainfo>
<titulo>Programacion Java</titulo>
<descripcion>web de programación en Java</descripcion>
<direcciones>http://www.euskalnet.net/jaoprogramador</direcciones>
<grupo>
<titulo>Programacion </titulo>
<id>1</id>
<direcciones>
<url>http://www.euskalnet.net/jaoprogramador</url>
<titulo>JAO</titulo>
<direccion>C/ Perez Galdos</direccion>
<idioma>castellano</idioma>
<valoracion>5</valoracion>
</direcciones>
</grupo>
</bookmark>

//Documento DTD:Ej1.dtd

<!ELEMENT bookmark (metainfo, titulo, descripcion?, (direcciones?|grupo))>
<!ELEMENT metainfo (titulo, autor, resumen?)>
<!ELEMENT titulo (#PCDATA)>
<!ELEMENT autor (#PCDATA)>
<!ELEMENT resumen (#PCDATA)>
<!ELEMENT titulo (#PCDATA)>
<!ELEMENT direcciones (#PCDATA)>
<!ELEMENT grupo (titulo, id, direcciones)>
<!ELEMENT titulo (#PCDATA)>
<!ELEMENT id (#PCDATA)>
<!ELEMENT direcciones (url,titulo,direccion?,idioma,valoracion)>
<!ELEMENT url (#PCDATA)>
<!ELEMENT titulo (#PCDATA)>
<!ELEMENT direccion (#PCDATA)>
<!ELEMENT idioma (castellano|ingles) #REQUIRED>
<!ELEMENT valoracion (1|2|3|4|5) #REQUIRED>
