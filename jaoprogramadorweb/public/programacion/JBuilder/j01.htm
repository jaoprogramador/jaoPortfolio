3.- pcirculo: aplicación que nos pide que introduzcamos el radio de un círculo y nos calcula el radio

package pcirculo;

import java.io.*;
public class circulo {
private static BufferedReader stdin = new BufferedReader(new InputStreamReader(System.in));
public static void main(String[] args)throws IOException {
//Declaracion de variables
final float PI=3.1416F; //constante PI
float area = 0;
int radio = 5; //radio del circulo
String input; //auxiliar para entrada System.out.print("\n Radio del circulo: ");
input = stdin.readLine(); //lo lee como un string
radio = Integer.parseInt(input); //lo convierte a entero
//area de un circulo
area=PI * radio*radio;
System.out.println("\n El area de un circulo de radio = " + radio + " es " + area); }
}

Character cx= new Character('x');

System.out.println("El caracter "+cx+" es minuscula");
codigo = c$.charValue();
System.out.println("Codigo de x = "+ codigo);

System.out.println("El caracter "+c$+" es mayuscula");
