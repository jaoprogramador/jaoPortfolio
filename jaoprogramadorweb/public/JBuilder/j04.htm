//cad[0]=cadena[3] ; ERROR aunque sean letras los tipos de datos son diferentes

num=4;

23.- pbiblioteca: Aplicación que mediante clases, objetos, funciones y menus nos permite introducir usuarios (nombre y dni) y libros, para luego verlos o modificarlos.

//en la clase biblioteca.java

package pbiblioteca;

import java.io.IOException;

public class biblioteca {
public static final short cantidad=3;//constante para el numero de usuarios totales
public static void main(String args[]) throws IOException {
menu menu2=new menu();//creamos el objeto MENU
short eleccion=0;


//creamos un array de usuarios y libros (USUARIOS y LIBROS) para 3 usuarios
usuario [] USUARIOS=new usuario[cantidad];
libro [] LIBROS=new libro[cantidad];

//reservamos memoria para los 3 usuarios
for(int i=0;i<cantidad;i++){
USUARIOS[i]=new usuario();
LIBROS[i]=new libro();
}

//////////////////////
do{
eleccion=menu2.Crear();//capturamos la opcion seleccionada y la almacenamos en eleccion
switch (eleccion) { //llamamos al menu y gestionamos la opcion pulsada
case 1: //Nuevo usuario
//introducir datos para el usuario
for(int i=0;i<cantidad;i++)
USUARIOS[i].setDatos();
break;
case 2: //Ver usuario
//Mostramos los datos de los 3 usuarios y los tres libros
for(int i=0;i<cantidad;i++)
USUARIOS[i].verDatos();
break;
case 3: //Nuevo libro
for(int i=0;i<cantidad;i++){
LIBROS[i].setTitulo();
LIBROS[i].setReservado();
}
break;
case 4: //Ver libro
for(int i=0;i<cantidad;i++)
LIBROS[i].verDatos();
break;
}
}while(eleccion>=1 && eleccion<=4);
System.out.println("Fin de la aplicacion");
System.out.println("====================");
}
}

do{
System.out.println("dni : ");
dni=stdin.readLine();

if(dni.length()==10 && dni.charAt(8)=='-'){//comprobamos q la longitud del dni sea de 10 y q la 8º sea un guión "-"
ok=true;//si ok esta a true es ke se kumplen todas las kondiciones
}
try{//TRY... CATCH es para tratar posibles errores q se produzca dentro de ellos
for(int i=0;i<dni.length()-2;i++){//recorremos todo el dni y comprobamos q todos sean numericos y q la ultima sea un caracter del alfabeto

ok=true;

else{ok=false;break;}
}
}catch(Exception e){}


}while (ok==false);


}
