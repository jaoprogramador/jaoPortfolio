6.- pcapital: Aplicación que calcula el pago mensual y total para un préstamo con distintas salidas de formatos.

package pcapital;

public class capital {
public static void main(String[] args) {
//Declaracion de variables
final int MESES_EN_UN_ANIO=12;
float cuantiaPrestamo=0.0F, interesAnual =0.0F, interesMensual;
int duracionPrestamo=1;
double pagoMensual = 0.0, pagoTotal=0.0;
int numeroPagos=0;
//parte ejecutiva del programa
System.out.println(" Este programa calcula el pago mensual y ");
System.out.println(" total para un prestamo");
//Entrada de datos
cuantiaPrestamo =100000;
interesAnual=3.5f;
duracionPrestamo=15;
//calculo de prestamos
interesMensual= interesAnual / 100.0f / MESES_EN_UN_ANIO;
numeroPagos=duracionPrestamo*MESES_EN_UN_ANIO;
pagoMensual=(cuantiaPrestamo*interesMensual)/(1-Math.pow(1/(1+interesMensual),numeroPagos));
pagoTotal=pagoMensual*numeroPagos;
// salida datos
System.out.println("\n Cuantia del prestamo: "+cuantiaPrestamo + " euros");
System.out.println(" Interes anual: "+interesAnual + " %");
System.out.println(" Duracion del prestamo (años) "+duracionPrestamo);
System.out.println(" Pago mensual "+pagoMensual + " euros");
System.out.println(" Cantidad total a pagar: "+pagoTotal + " euros");
}
}
